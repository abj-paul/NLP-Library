#include "lib_vector.h"
#include "lib_string.h"

int main(){
  abj::Vector<abj::String> x;
  abj::String a("Abhi");
  abj::String b("Jit");
  abj::String c("Paul");
  
  x.push(a);
  x.push(b);
  x.push(c);

  for(int i=0; i<x.size(); i++) x[i].print();
  x.reverse();

  for(int i=0; i<x.size(); i++) x[i].print();
  return 0;
}
